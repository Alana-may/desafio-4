    async function fetchData(endpoint, params = {}) {

        // Adicionar chave de API se necess√°ria: headers: { 'Authorization': 'Bearer SEU_TOKEN_API' }

        try {
            const response = await fetch(url);
            if (!response.ok) {
                if (response.status === 404) console.warn(`Dados n√£o encontrados para: ${url}`);
                else console.error(`Erro HTTP: ${response.status} para ${url}`);
                return null; // Tratar como dados n√£o encontrados ou erro espec√≠fico
